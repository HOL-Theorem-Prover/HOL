% Keep the PDF citation style in the same family as the mdbook
% side, which renders via chicago-author-date.csl through pandoc
% citeproc as `(Author Year)` and `(Author Year, loc)`.
%
%   [authoryear] selects the author-year mode (vs natbib's default
%               [numbers] which would give "[1]" style cites).
%   [round]     emits round parens instead of natbib's default
%               square brackets, matching the mdbook's parens.
%   \cite ≡ \citep ensures every `\cite{key}` in the .smd source
%               renders parenthetically -- by default natbib aliases
%               `\cite` to `\citet` (text-style "Author (Year)") on
%               bare cites but to `\citep` on cites with locators,
%               producing a mixed look across the document.  The
%               .smd source predates this style choice, so a
%               uniform \citep is the closest match to the mdbook.
\usepackage[authoryear,round]{natbib}
\let\cite\citep

% holtexbasic.sty pulls in `underscore` without options.  Without
% `[strings]`, underscore activates `_` at \begin{document} but
% skips running its `\UnderscoreCommands` protection list, so the
% active `_` breaks `\csname r@...\endcsname` lookups on label
% names like `the-stateful-simpset-srw_ss` that pandoc emits.
% Pre-load underscore with `[strings]` so the protections run.
% holtexbasic's later unoptioned \usepackage{underscore} is then
% a no-op (already loaded).
%
% Then pre-declare `\US_cite` so underscore's `\do\cite` step
% skips its protection wrapper for `\cite`.  Underscore wraps
% optional-arg-accepting commands such that `\cite[#1]{#2}` puts
% `\cite[#1]` inside its own group, breaking natbib's parse and
% leaving `[` as the (undefined) cite key.  None of the cite
% keys this manual uses contain `_`, so we don't need cite-
% argument protection in any case.
\expandafter\def\csname US_cite\endcsname{}
